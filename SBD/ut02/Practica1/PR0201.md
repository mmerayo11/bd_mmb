# PR0201: Cadenas en Redis
## 1. Trabajo con redis-cli
Estos comandos se ejecutan directamente en la interfaz de línea de comandos de Redis.
```
# Establecer valores para claves
Set "usuario:nombre" "Mario"  
Set "usuario:apellido" "Merayo"

# Obtener valores de claves
get "usuario:nombre"  
get "usuario:apellido"

# Establecer y obtener un email
Set "usuario:email" "MarioMerayo@hotmail.com"  
get "usuario:email"

# Sobrescribir un valor
Set "usuario:nombre" "MARIO"

# Trabajar con contadores
Set "contador:visitas" 0  
INCR "contador:visitas"  
INCR "contador:visitas"  
INCR "contador:visitas"  
INCR "contador:visitas"  
DECR "contador:visitas"  
DECR "contador:visitas"  

# Establecer un mensaje con expiración (Time To Live - TTL)
Set "mensaje" "Bienvenido a Redis" EX 60

# Eliminar claves
Del "usuario:apellido"
Del "usuario:nombre" "usuario:email" "contador:visitas" "mensaje"
```

## 2. Trabajo con la librería redis-py
Este es un ejemplo de cómo interactuar con Redis usando el cliente oficial de Python.
```
# Instalar la librería (si no está instalada)
!pip install redis

import redis

# Conectarse a Redis
r = redis.Redis(
    host='redis',
    port=6379,
    db=0,
    decode_responses=True  # Decodifica las respuestas de bytes a strings
)

# --- Operaciones básicas de Set/Get ---
r.set("app:version", "1.0")
print(f"Versión actual: {r.get('app:version')}")

# Sobrescribir un valor
r.set("app:version", "1.1")
print(f"Versión actualizada: {r.get('app:version')}")

# --- Contadores (Incrementar/Decrementar) ---
r.set("contador:descargas", 0)
print(f"Descargas iniciales: {r.get('contador:descargas')}")

r.incrby("contador:descargas", 5)
print(f"Descargas después de +5: {r.get('contador:descargas')}")

r.decrby("contador:descargas", 2)
print(f"Descargas después de -2: {r.get('contador:descargas')}")

# --- Expiración de claves (TTL) ---
r.set("app:estado", "activo")
r.set("app:estado", "mantenimiento")
r.set("mensaje:bienvenida", "Hola alumno")

# Establecer que la clave "app:estado" expire en 30 segundos
r.expire("app:estado", 30)

# Comprobar si una clave existe
print(f"¿Existe 'app:estado'? {r.exists('app:estado')}")

# --- Eliminar claves ---
r.delete("app:version")

if not r.exists("app:version"):
    print("app:version ha sido eliminada con éxito")

```
